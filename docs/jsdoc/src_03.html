 80 /**
 81  * Constructs a suppression filter from a bare object with suitable contents.
 82  * For use in deserializing filters from JSON. Throws a string exception if an
 83  * invalid filter is encountered.
 84  * @param {!Object} bareFilter The object representing the filter. Must contain
 85  *     an 'opcode' field with a string indicating the filter type.
 86  *     Other supported field names and types are dependent on the filter type.
 87  * @return {!bidichecker.FilterFactory.ComposableFilter} The new filter.
 88  * @private
 89  */
 90 bidichecker.FilterFactory.constructFilter_ = function(bareFilter) {
 91   var opcode = bareFilter['opcode'];
 92   switch (opcode) {
 93     case 'AND':
 94       return new bidichecker.FilterFactory.AndFilter_(bareFilter);
 95 
 96     case 'AT_TEXT':
 97       return new bidichecker.FilterFactory.AtTextFilter_(bareFilter);
 98 
 99     case 'AT_TEXT_REGEXP':
100       return new bidichecker.FilterFactory.AtTextRegexpFilter_(bareFilter);
101 
102     case 'FOLLOWED_BY_TEXT':
103       return new bidichecker.FilterFactory.FollowedByTextFilter_(bareFilter);
104 
105     case 'FOLLOWED_BY_TEXT_REGEXP':
106       return new bidichecker.FilterFactory.FollowedByTextRegexpFilter_(
107           bareFilter);
108 
109     case 'LOCATION_CLASS':
110       return new bidichecker.FilterFactory.LocationClassFilter_(bareFilter);
111 
112     case 'LOCATION_CLASS_REGEXP':
113       return new bidichecker.FilterFactory.LocationClassRegexpFilter_(
114           bareFilter);
115 
116     case 'LOCATION_ID':
117       return new bidichecker.FilterFactory.LocationIdFilter_(bareFilter);
118 
119     case 'LOCATION_ID_REGEXP':
120       return new bidichecker.FilterFactory.LocationIdRegexpFilter_(
121           bareFilter);
122 
123     case 'LOCATION_XPATH':
124       return new bidichecker.FilterFactory.LocationXpathFilter_(
125           bareFilter);
126 
127     case 'NOT':
128       return new bidichecker.FilterFactory.NotFilter_(bareFilter);
129 
130     case 'OR':
131       return new bidichecker.FilterFactory.OrFilter_(bareFilter);
132 
133     case 'PRECEDED_BY_TEXT':
134       return new bidichecker.FilterFactory.PrecededByTextFilter_(bareFilter);
135 
136     case 'PRECEDED_BY_TEXT_REGEXP':
137       return new bidichecker.FilterFactory.PrecededByTextRegexpFilter_(
138           bareFilter);
139 
140     case 'SEVERITY':
141       return new bidichecker.FilterFactory.SeverityFilter_(bareFilter);
142 
143     case 'TYPE':
144       return new bidichecker.FilterFactory.TypeFilter_(bareFilter);
145 
146     default:
147       throw 'Unknown filter opcode: \'' + opcode + '\'';
148   }
149 };

152 /**
153  * Deserializes a list of suppression filters from a JSON-format string.
154  * Throws a string exception if an unknown filter is encountered.
155  * @param {string=} opt_filtersJson The input string, if provided.
156  * @return {Array.<!bidichecker.Filter>} The error suppression filters.
157  */
158 bidichecker.FilterFactory.readFiltersFromJson = function(opt_filtersJson) {
159   if (!opt_filtersJson) {
160     return null;
161   }
162 
163   var bareFilters =
164       /** @type {Array.<!Object>} */ (goog.json.parse(opt_filtersJson));
165   return goog.array.map(bareFilters, function(bareFilter) {
166     return bidichecker.FilterFactory.constructFilter_(bareFilter);
167   });
168 };

413 /**
414  * Checks that a required field exists in an object representing a filter, or
415  * else throws a string exception.
416  * @param {!Object} bareObject The object representing the filter.
417  * @param {string} field The name of the field.
418  * @private
419  */
420 bidichecker.FilterFactory.checkParamExists_ = function(bareObject, field) {
421   if (!(field in bareObject)) {
422     throw 'No \'' + field + '\' parameter found for \'' +
423         bareObject['opcode'] + '\' filter';
424   }
425 };

428 /**
429  * Checks that a required field of a given type exists in an object representing
430  * a filter, or else throws a string exception.
431  * @param {!Object} bareObject The object representing the filter.
432  * @param {string} field The name of the field.
433  * @param {string} type The required type of the field.
434  * @private
435  */
436 bidichecker.FilterFactory.checkParam_ = function(bareObject, field, type) {
437   bidichecker.FilterFactory.checkParamExists_(bareObject, field);
438   if (typeof bareObject[field] != type) {
439     throw 'Wrong type for \'' + field + '\' parameter of \'' +
440         bareObject['opcode'] + '\' filter; expected ' + type;
441   }
442 };

471 /**
472  * Reads a regular expression field from an object representing a filter.
473  * The field contents must be a string or a RegExp object.
474  * @param {!Object} bareObject The object representing the filter.
475  * @param {string} fieldName The name of the field.
476  * @return {!RegExp} A regular expression anchored at both ends of the string.
477  * @private
478  */
479 bidichecker.FilterFactory.getRegexpParam_ = function(bareObject, fieldName) {
480   bidichecker.FilterFactory.checkParamExists_(bareObject, fieldName);
481   var field = bareObject[fieldName];
482   var pattern;
483   if (typeof field == 'string') {
484     pattern = /** @type {string} */ (field);
485   } else if (field instanceof RegExp) {
486     pattern = field.source;
487   } else {
488     throw 'Wrong type for \'' + fieldName + '\' parameter of \'' +
489         bareObject['opcode'] + '\' filter; expected string or RegExp';
490   }
491   return new RegExp('^(' + pattern + ')$');
492 };

495 /**
496  * Reads a field representing a subfilter from an object representing a filter.
497  * The field contents can either be a bare object representing a serialized
498  * filter, or an object supporting an isSuppressed() method.
499  * @param {!Object} bareObject The object representing the filter.
500  * @param {string} field The name of the field.
501  * @return {!bidichecker.FilterFactory.ComposableFilter} A filter constructed
502  *     from the field's contents.
503  * @private
504  */
505 bidichecker.FilterFactory.getFilterParam_ = function(
506     bareObject, field) {
507   bidichecker.FilterFactory.checkParam_(bareObject, field, 'object');
508 
509   // Recursively build any subfilters referred to by this one.
510   var subfilterObject = bareObject[field];
511   if (subfilterObject instanceof bidichecker.FilterFactory.ComposableFilter) {
512     // Found a constructed filter object.
513     return subfilterObject;
514   } else if (typeof subfilterObject['opcode'] == 'string') {
515     // Found a raw filter object.
516     return bidichecker.FilterFactory.constructFilter_(subfilterObject);
517   } else {
518     throw 'Can\'t make a filter out of the \'' + field + '\' parameter of \'' +
519         bareObject['opcode'] + '\' filter';
520   }
521 };
